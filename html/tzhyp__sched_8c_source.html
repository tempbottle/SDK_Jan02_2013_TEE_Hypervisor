49 void tzhyp_schedule_guest(void)
50 {
51  static struct system_context *boundary = &ns_world[0] + GUESTS_NO;
52  struct system_context *current, *next;
53 #ifdef SCHEDULE_HIGH_PRIORITY_GUEST
54  static u32 count = 0, low_priority_guest_scheduled = 0;
55 #endif
56 
57 #ifdef SW_OTZHYP_DEBUG
58  sw_printf("tzhyp_sched_guest\n");
59 #endif
60 
61  current = ns_sys_current;
62 #ifdef SCHEDULE_HIGH_PRIORITY_GUEST
63  if(current->guest_no != HIGH_PRIORITY_GUEST)
64  low_priority_guest_scheduled = 1;
65 
66  if(low_priority_guest_scheduled &&
67  is_guest_irq_active(current->guest_no)) {
68  /* Schedule the low priorty guest @ 800 milli seconds interval
69  40ms * 20 = 800 milli seconds */
70  if(count < 20) {
71  count ++;
72  return current->guest_no;
73  }
74  else {
75  count = 0;
76  }
77  }
78 #endif
79 
80  next = ns_sys_current + 1;
81  if (next >= boundary) /* reset to index 0 */
82  next = ns_world;
83 #ifdef SW_OTZHYP_DEBUG
84  sw_printf("current = %x, next = %x\n", current, next);
85 #endif
86 
98  ns_sys_current = next;
99  tzhyp_sysregs_switch(&current->sysctxt_cp15, &next->sysctxt_cp15);
100  tzhyp_device_switch(current, next);
101 }
